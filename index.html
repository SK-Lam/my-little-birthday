<script>

  document.getElementById("friendName").innerText = “包包蟲”;


  const editor = document.getElementById("pixelEditor");

  const colorPicker = document.getElementById("colorPicker");

  const savedArt = document.getElementById("savedArt");

  const size = 16;

  let mode = "draw";

  let drawing = false;

  const cells = [];


  function createBoard() {

    for (let i = 0; i < size * size; i++) {

      const cell = document.createElement("div");

      cell.className = "pixel";

      cell.dataset.index = i;


      cell.addEventListener("mousedown", () => {

        drawing = true;

        paint(cell);

      });


      cell.addEventListener("mouseenter", () => {

        if (drawing) paint(cell);

      });


      cell.addEventListener("touchstart", (e) => {

        e.preventDefault();

        paint(cell);

      }, { passive: false });


      editor.appendChild(cell);

      cells.push(cell);

    }

  }


  function paint(cell) {

    if (mode === "draw") {

      cell.style.background = colorPicker.value;

    } else if (mode === "erase") {

      cell.style.background = "#ffffff";

    }

  }


  function setMode(newMode) {

    mode = newMode;

  }


  function clearBoard() {

    cells.forEach(cell => cell.style.background = "#ffffff");

  }


  document.body.addEventListener("mouseup", () => drawing = false);


  function drawPattern(pattern, colorMap) {

    clearBoard();

    pattern.forEach((row, y) => {

      row.forEach((v, x) => {

        if (v !== 0) {

          const idx = y * size + x;

          cells[idx].style.background = colorMap[v];

        }

      });

    });

  }

  }

  function getBoardData() {

    return cells.map(c => c.style.background || "#ffffff");

  }


  function saveArt() {

    const data = getBoardData();

    const canvas = document.createElement("canvas");

    const scale = 8;

    canvas.width = size * scale;

    canvas.height = size * scale;

    const ctx = canvas.getContext("2d");


    for (let y = 0; y < size; y++) {

      for (let x = 0; x < size; x++) {

        const color = data[y * size + x] || "#ffffff";

        ctx.fillStyle = color;

        ctx.fillRect(x * scale, y * scale, scale, scale);

      }

    }


    const img = document.createElement("img");

    img.src = canvas.toDataURL("image/png");

    img.className = "mini-art";

    savedArt.prepend(img);

  }


  function toggleMusic() {

    const music = document.getElementById("bgMusic");

    if (music.paused) {

      music.play();

    } else {

      music.pause();

    }

  }


  function createFloatingEmoji() {

    const icons = ["💖", "⭐", "🧁", "🎂", "🍓", "☁️"];

    const item = document.createElement("div");

    item.className = "floating";

    item.innerText = icons[Math.floor(Math.random() * icons.length)];

    item.style.left = Math.random() * 100 + "vw";

    item.style.animationDuration = (5 + Math.random() * 6) + "s";

    item.style.fontSize = (20 + Math.random() * 20) + "px";

    document.body.appendChild(item);


    setTimeout(() => item.remove(), 12000);

  }
